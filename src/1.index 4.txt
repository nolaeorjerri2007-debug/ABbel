  <script>
    document.addEventListener("DOMContentLoaded", () => {
      
      const intentInput = document.getElementById('intentInput');
      const charCounter = document.getElementById('charCounter');
      const btnGenerate = document.getElementById('btnGenerate');
      const navBtns = document.querySelectorAll('.btn-nav');
      const dialCards = document.querySelectorAll('.dial-card');
      const slotItems = document.querySelectorAll('.slot-item');

      // 1. 实时字符计数器
      intentInput.addEventListener('input', (e) => {
        const len = e.target.value.length;
        const formattedLen = String(len).padStart(4, '0');
        charCounter.textContent = len > 0 ? `TYPING_ ${formattedLen} / 2000` : `READY_ 0000 / 2000`;
        charCounter.style.opacity = len > 0 ? '1' : '0.5';
      });

      // 支持 CMD/CTRL + Enter 快速提交
      intentInput.addEventListener('keydown', (e) => {
        if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {
          btnGenerate.click();
        }
      });

      // 2. 导航栏点按高亮联动
      navBtns.forEach(btn => {
        btn.addEventListener('click', function() {
          // 熄灭所有导航
          navBtns.forEach(b => {
            b.classList.remove('active');
            const led = b.querySelector('.led');
            led.classList.remove('on', 'red');
            led.classList.add('off');
          });
          // 点亮当前导航
          this.classList.add('active');
          const led = this.querySelector('.led');
          led.classList.remove('off');
          led.classList.add('on', 'red');
        });
      });

      // 3. 预设卡片载入模拟
      dialCards.forEach(card => {
        card.addEventListener('click', function() {
          const title = this.querySelector('.dial-title').textContent;
          // 将模板指令拍入输入框
          intentInput.value = `[预设载入: ${title}] 请帮我针对以下产品写一段...`;
          intentInput.dispatchEvent(new Event('input')); // 手动触发计数器更新
          intentInput.focus();
        });
      });

      // 4. 专属记忆槽载入模拟
      slotItems.forEach(slot => {
        slot.addEventListener('click', function() {
          const title = this.textContent.trim();
          intentInput.value = `[专属记忆位: ${title}] 按照此配置帮我处理以下内容...`;
          intentInput.dispatchEvent(new Event('input'));
          intentInput.focus();
        });
      });

      // 5. 执行解压 (提交验证)
      btnGenerate.addEventListener('click', function() {
        if (intentInput.value.trim() === '') {
          // 报错：空状态震动与闪烁
          charCounter.textContent = 'SYS_ERR: 请输入解压指令！';
          charCounter.style.color = 'var(--lcd-text-danger)';
          charCounter.style.opacity = '1';
          intentInput.focus();
          
          setTimeout(() => {
            charCounter.textContent = 'READY_ 0000 / 2000';
            charCounter.style.color = 'var(--lcd-text-bright)';
            charCounter.style.opacity = '0.5';
          }, 2000);
          return;
        }

        // 模拟执行加载动画
        const originalText = this.innerHTML;
        this.innerHTML = '解析并连接引擎...';
        this.style.pointerEvents = 'none';
        this.style.opacity = '0.8';

        setTimeout(() => {
          this.innerHTML = originalText;
          this.style.pointerEvents = 'auto';
          this.style.opacity = '1';
          
          // 模拟页面跳转 (如果是在同一文件夹下，您可以直接替换为 window.location.href = 'workspace.html')
          alert('✔ 指令解析完毕，即将进入【工作台】...');
        }, 800);
      });

      // 页面加载完毕后自动聚焦输入框
      intentInput.focus();
    });
  </script>
