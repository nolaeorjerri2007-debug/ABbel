<script>
  document.addEventListener("DOMContentLoaded", () => {
    
    // 骨架屏流水灯
    const dots = document.querySelectorAll('.skeleton-loader .dot');
    let currentDot = 0;
    setInterval(() => {
      dots.forEach(dot => dot.classList.remove('active'));
      dots[currentDot].classList.add('active');
      if(currentDot > 0) dots[currentDot - 1].classList.add('active');
      if(currentDot < dots.length - 1) dots[currentDot + 1].classList.add('active');
      currentDot++;
      if (currentDot >= dots.length) currentDot = 0;
    }, 150);

    // 追加上下文抽屉
    const contextDrawer = document.getElementById('contextDrawer');
    document.getElementById('btnAppendContext').addEventListener('click', () => {
      contextDrawer.classList.add('open');
      setTimeout(() => contextDrawer.querySelector('.context-input').focus(), 300);
    });
    document.getElementById('btnCloseDrawer').addEventListener('click', () => contextDrawer.classList.remove('open'));
    window.submitContext = () => {
      const textarea = contextDrawer.querySelector('.context-input');
      if(textarea.value.trim() !== "") textarea.value = ""; 
      contextDrawer.classList.remove('open');
    };

    // ==========================================
    // 全新重构：批量结算 (购物车) & 全自动引擎
    // ==========================================
    const sliders = document.querySelectorAll('.slider-item');
    const applyBar = document.getElementById('applyBar');
    const targetSegment = document.getElementById('targetSegment');
    const previewNode = document.getElementById('previewNode');
    const previewLine = document.getElementById('previewLine');
    const controlContent = document.getElementById('controlContent');
    const soloOverlay = document.getElementById('soloOverlay');
    const autoApplyToggle = document.getElementById('autoApplyToggle');
    const batchCountText = document.getElementById('batchCountText');
    const tokenEstText = document.getElementById('tokenEstText');

    // 核心状态管理
    const stagedParams = new Set(); // “购物车”：存放已偏移的滑块节点
    let isAutoApplyOn = false;
    let autoApplyTimer = null;
    let soloTimer = null;
    const originalText = targetSegment.innerHTML;

    // 参数词典与 Mock
    const paramDict = {
      intimacy: { low: "语气将变得客观克制...", high: "语气将变得亲密熟络...", mock: "最近有很多新手妈妈私信我，说秋冬换季宝宝皮肤干燥。这款修护霜加入了5%的泛醇和神经酰胺，修护效果有口皆碑。" },
      info_density: { low: "文本将变得通俗易懂...", high: "将加入更多硬核参数...", mock: "核心配方为 5% 泛醇（B5）复合神经酰胺（NP/AP/EOP），能显著降低 TEWL（经表皮失水率），重塑受损屏障。" },
      arousal: { low: "情绪将趋于平缓...", high: "用词将极具煽动性...", mock: "天呐真的绝了！这瓶简直是来报恩的吧😭！满满的 B5 和神经酰胺，上脸就像给宝宝穿了隐形保暖衣！" },
      urgency: { low: "转为客观陈述...", high: "加入强烈的行动号召...", mock: "群里的宝妈们别犹豫了，趁现在有活动赶紧囤！宝宝烂脸可等不起，早用早退红，手慢无！" },
      jargon_density: { low: "大白话表达...", high: "增加垂类黑话...", mock: "主打一个仿生皮脂膜体系，B5 协同神经酰胺强效打底，退红维稳一把抓，屏障受损肌直接闭眼入。" },
      directness: { low: "表达更开门见山...", high: "增加含蓄感...", mock: "换季风大，有些修护霜到底好不好用，看看大家的空瓶就知道了。成分配方摆在这里，懂的自然懂。" }
    };

    // 全自动引擎开关监听
    autoApplyToggle.addEventListener('change', (e) => {
      isAutoApplyOn = e.target.checked;
      if (isAutoApplyOn && stagedParams.size > 0) {
        // 如果开启时已有暂存参数，立即自动应用
        applyBar.classList.remove('visible');
        targetSegment.classList.add('auto-applying');
        autoApplyTimer = setTimeout(() => applyChanges(), 1000);
      } else {
        targetSegment.classList.remove('auto-applying');
        clearTimeout(autoApplyTimer);
        if(!isAutoApplyOn && stagedParams.size > 0) applyBar.classList.add('visible'); // 关掉自动时，恢复底部 Bar
      }
    });

    // 重置 / 取消
    window.cancelChanges = () => {
      sliders.forEach(slider => {
        if(!slider.dataset.param) return;
        const origVal = slider.dataset.original;
        updateSliderVisual(slider, origVal);
        slider.classList.remove('staged');
      });
      stagedParams.clear();
      targetSegment.classList.remove('staged', 'auto-applying');
      targetSegment.innerHTML = originalText;
      applyBar.classList.remove('visible');
      previewNode.style.display = 'none';
      previewLine.style.display = 'none';
      clearTimeout(autoApplyTimer);
    };

    // 应用变更
    window.applyChanges = () => {
      targetSegment.classList.remove('staged', 'auto-applying');
      stagedParams.forEach(s => {
        s.dataset.original = s.querySelector('.slider-val').innerText;
        s.classList.remove('staged');
      });
      stagedParams.clear();
      applyBar.classList.remove('visible');
      previewNode.classList.remove('branch');
      previewNode.classList.add('active'); 
      previewLine.style.borderStyle = 'solid'; 
    };

    document.addEventListener('keydown', (e) => { if (e.key === 'Escape' && stagedParams.size > 0) cancelChanges(); });
    soloOverlay.addEventListener('click', () => { controlContent.classList.remove('has-solo'); sliders.forEach(s => s.classList.remove('is-solo')); });

    const updateSliderVisual = (slider, percentVal) => {
      slider.querySelector('.slider-fill').style.width = (percentVal * 100).toFixed(1) + '%';
      slider.querySelector('.slider-thumb').style.left = (percentVal * 100).toFixed(1) + '%';
      slider.querySelector('.slider-val').textContent = Number(percentVal).toFixed(2);
    };

    sliders.forEach(slider => {
      const track = slider.querySelector('.slider-track');
      const thumb = slider.querySelector('.slider-thumb');
      const paramName = slider.dataset.param;
      if(!track || !thumb) return;

      let isDragging = false;

      const handleMove = (e) => {
        if (!isDragging) return;
        const rect = track.getBoundingClientRect();
        let percent = Math.max(0, Math.min(1, (e.clientX - rect.left) / rect.width));
        updateSliderVisual(slider, percent);
        if(paramName && paramDict[paramName]) {
          slider.querySelector('.slider-tooltip').innerText = percent > 0.5 ? paramDict[paramName].high : paramDict[paramName].low;
        }
      };

      thumb.addEventListener('mousedown', (e) => {
        isDragging = true;
        slider.classList.add('is-dragging');
        soloTimer = setTimeout(() => {
          if(isDragging) {
            sliders.forEach(s => s.classList.remove('is-solo'));
            slider.classList.add('is-solo');
            controlContent.classList.add('has-solo');
            if(navigator.vibrate) navigator.vibrate(50);
          }
        }, 600);
      });

      document.addEventListener('mousemove', handleMove);

      document.addEventListener('mouseup', () => {
        if(!isDragging) return;
        isDragging = false;
        slider.classList.remove('is-dragging');
        clearTimeout(soloTimer);

        const currentVal = parseFloat(slider.querySelector('.slider-val').innerText);
        const originalVal = parseFloat(slider.dataset.original);

        // 判定偏差入列/出列
        if (Math.abs(currentVal - originalVal) > 0.15) {
          slider.classList.add('staged');
          stagedParams.add(slider);
        } else {
          slider.classList.remove('staged');
          stagedParams.delete(slider);
          updateSliderVisual(slider, originalVal); // 弹回
        }

        // 统一处理“购物车”结算视图
        if (stagedParams.size > 0) {
          targetSegment.classList.add('staged');
          previewNode.style.display = 'flex';
          previewLine.style.display = 'block';

          // 决定幽灵层文案 (单个具体化，多个抽象化)
          if (stagedParams.size === 1) {
            const firstParam = Array.from(stagedParams)[0].dataset.param;
            targetSegment.innerText = paramDict[firstParam]?.mock || originalText;
          } else {
            targetSegment.innerText = "（检测到多维参数矩阵已变更，系统正在根据最新的权重组合，准备为您重构最优文案表达...）";
          }

          // 分流引擎逻辑
          if (isAutoApplyOn) {
            applyBar.classList.remove('visible');
            targetSegment.classList.add('auto-applying');
            clearTimeout(autoApplyTimer);
            autoApplyTimer = setTimeout(() => applyChanges(), 1500); // 1.5s 防抖自动提交
          } else {
            targetSegment.classList.remove('auto-applying');
            batchCountText.innerText = `${stagedParams.size} 个参数已偏移`;
            tokenEstText.innerText = `≈ 消耗 ${10 + stagedParams.size * 15} Tokens`;
            applyBar.classList.add('visible');
          }
        } else {
          cancelChanges(); // 如果所有滑块都拉回了原位，自动撤销态
        }
      });

      track.addEventListener('mousedown', (e) => { if(e.target !== thumb) { isDragging = true; handleMove(e); } });
    });
  });
</script>
